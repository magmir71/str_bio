def checklipinsky(mol):
    l = [0] * 4
    if Lipinksy.NumHDonors(mol)<=5:
        l[0] = 1
    if Lipinksy.NumHAcceptors(mol)<=10:
        l[1] = 1
    if Lipinksy.rdMolDescriptors.CalcExactMolWt(mol)<500:
        l[2] = 1
    if Lipinksy.rdMolDescriptors.CalcCrippenDescriptors(mol)[0]<=5:
        l[3] = 1
    
    return sum(l)>=3
